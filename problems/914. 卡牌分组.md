# 先求计数，再求最大公约数
### 解题思路
* 1、对``deck``中所有数，求得其计数``m``
* 2、对所有计数``m``，求其最大公约数，看看是否``>=2``

### 代码

```golang

func hasGroupsSizeX(deck []int) bool {
	m := make(map[int]int)
	max := 0
	for _, v := range deck {
		m[v]++
		if v > max {
			max = v
		}
	}
	g := 0
	for i := 0; i <= max; i++ {
		if m[i] > 0 {
			if g == 0 {
				g = m[i]
			} else {
				g = gcd(g, m[i])
			}
		}
	}
	return g >= 2
}

func gcd(x, y int) int {
	if y == 0 {
		return x
	}
	return gcd(y, x%y)
}
```
