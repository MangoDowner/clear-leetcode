# 170. 两数之和 III - 数据结构设计
### 代码
```golang
type TwoSum struct {
	Freq map[int]int
}

func Constructor() TwoSum {
	return TwoSum{map[int]int{}}
}

func (this *TwoSum) Add(number int) {
	this.Freq[number]++
}

func (this *TwoSum) Find(value int) bool {
	for num := range this.Freq {
		other := value - num
		// 情况1
		if other == num && this.Freq[other] > 1 {
			return true
		}
		// 情况2
		if other != num && this.Freq[other] > 0 {
			return true
		}
	}
	return false
}
```
