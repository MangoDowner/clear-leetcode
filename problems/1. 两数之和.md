> 原题链接：[1. 两数之和](https://leetcode-cn.com/problems/two-sum/)
### 代码
```golang
func twoSum(nums []int, target int) []int {
	m := make(map[int]int)
	for k, v := range nums {
		m[v] = k
	}
	for k, v:= range nums {
		if v1, ok := m[target - v]; ok && k != v1 {
			return []int{k, v1}
		}
	}
	return []int{}
}
```
