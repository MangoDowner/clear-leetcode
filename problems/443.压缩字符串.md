func compress(chars []byte) int {
	anchor, write := 0, 0 // 节点和写指针位置
	for read, v := range chars {
		// 如果已经是最后一项,或者与后一项不同
		if read == len(chars)-1 || chars[read+1] != v {
			chars[write] = chars[anchor]
			write++ // 写指针后移
			if read > anchor {
				for _, digit := range strconv.Itoa(read - anchor + 1) {
					chars[write] = byte(digit)
					write++
				}
			}
			anchor = read + 1
		}
	}
	return write
}
