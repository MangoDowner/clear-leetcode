# 递归法和循环法
## 递归法
### 代码

```golang
func reverseList(head *ListNode) *ListNode {
	result := new(ListNode)
	result = nil
	return loop(head, result)
}

func loop(head *ListNode, result *ListNode) *ListNode {
	if head == nil {
		return result
	}
	result = &ListNode{head.Val, result}
	return loop(head.Next, result)
}
```
下面是官方的写法golang版
```golang
func reverseList(head *ListNode) *ListNode {
	if head == nil || head.Next == nil {
		return head
	}
	result := reverseList(head.Next)
	head.Next.Next= head
	head.Next = nil
	return result
}
```
## 循环法
### 代码

```golang
func reverseList(head *ListNode) *ListNode {
	result := new(ListNode)
	result = nil
	for head != nil {
		result = &ListNode{head.Val, result}
		head = head.Next
	}
	return result
}
```
