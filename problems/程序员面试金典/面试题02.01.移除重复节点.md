# 面试题 02.01. 移除重复节点
> 原题链接：[面试题 02.01. 移除重复节点](https://leetcode-cn.com/problems/remove-duplicate-node-lcci/)

### 代码
c#
```csharp
/**
 * Definition for singly-linked list.
 * public class ListNode {
 *     public int val;
 *     public ListNode next;
 *     public ListNode(int x) { val = x; }
 * }
 */
public class Solution {
    public ListNode RemoveDuplicateNodes(ListNode head)
    {
        if (head?.next == null)
        {
            return head;
        }

        for (var outer = head; outer != null; outer = outer.next)
        {
            for (ListNode inner = outer.next, innerPre = outer; inner != null;)
            {
                if (outer.val == inner.val)
                {
                    innerPre.next = inner.next;
                    inner = inner.next;
                }
                else
                {
                    innerPre = inner;
                    inner = inner.next;
                }
            }
        }

        return head;
    }
}
```

```csharp
/**
 * Definition for singly-linked list.
 * public class ListNode {
 *     public int val;
 *     public ListNode next;
 *     public ListNode(int x) { val = x; }
 * }
 */
public class Solution {
    public ListNode RemoveDuplicateNodes(ListNode head)
    {
        if (head?.next == null)
        {
            return head;
        }

        head.next = RemoveDuplicateNodes(head.next);
        ListNode pre = head, cur = head.next;
        while (cur != null)
        {
            if (head.val == cur.val)
            {
                pre.next = cur.next;
                cur = cur.next;
            }
            else
            {
                pre = pre.next;
                cur = cur.next;
            }
        }

        return head;
    }
}
```
