> 原题链接：[752. 打开转盘锁](https://leetcode-cn.com/problems/open-the-lock/)
## BFS
### 解题思路
* 1、所有密码状态，分为4个位置的向上向下操作
* 2、从``0000``出发，将所有变化加入队列，看看最终能不能到达``target``
* 3、记得除去``deadends``里面状态，还有已经经过的状态
### 代码
```golang
func openLock(deadends []string, target string) int {
	mDead, mSeen := make(map[string]bool), map[string]bool{"0000": true}
	queue, step := []string{"0000"}, 0
	for _, v := range deadends {
		mDead[v] = true
	}
	for len(queue) != 0 {
		size := len(queue)
		for i := 0; i < size; i++ {
			head := queue[i]
			// 是否到达终点
			if head == target {
				return step
			}
            if mDead[head] {
				continue
			}
			for j := 0; j < 4; j++ {
				plus, minus := plusOne(head, j), minusOne(head, j)
				if !mSeen[plus] { 
					queue = append(queue, plus)
					mSeen[plus] = true

				}
				if !mSeen[minus] {
					queue = append(queue, minus)
					mSeen[minus] = true
				}
			}
		}
		step, queue = step+1, queue[size:]
	}
	return -1
}

func plusOne(s string, index int) string {
	sRune := []rune(s)
	if sRune[index] == '9' {
		sRune[index] = '0'
	} else {
		sRune[index] += 1
	}
	return string(sRune)
}

func minusOne(s string, index int) string {
	sRune := []rune(s)
	if sRune[index] == '0' {
		sRune[index] = '9'
	} else {
		sRune[index] -= 1
	}
	return string(sRune)
}
```
## 双端BFS
### 解题思路
上面的做法是从上往下遍历，我们还可以从目标态同时往中间遍历，如果两边相遇了，就OK
### 代码
```golang
func openLock(deadends []string, target string) int {
	mDead, mSeen := make(map[string]bool), map[string]bool{"0000": true}
	q1, q2, step := make(map[string]bool), make(map[string]bool), 0
	q1["0000"], q2[target] = true, true
	for _, v := range deadends {
		mDead[v] = true
	}
	for len(q1) != 0 && len(q2) != 0 {
		temp := make(map[string]bool)
		for k := range q1 {
			if mDead[k] {
				continue
			}
			if q2[k] {
				return step
			}
			mSeen[k] = true
			for j := 0; j < 4; j++ {
				plus, minus := plusOne(k, j), minusOne(k, j)
				if !mSeen[plus] && !mDead[plus] { // 没用过，且不是死亡密码，再加进去
					temp[plus] = true
				}
				if !mSeen[minus] && !mDead[minus] {
					temp[minus] = true
				}
			}
		}
		step++
		// 这里交换 q1 q2，下一轮 while 就是扩散 q2
		q1, q2 = q2, temp
	}
	return -1
}

func plusOne(s string, index int) string {
	sRune := []rune(s)
	if sRune[index] == '9' {
		sRune[index] = '0'
	} else {
		sRune[index] += 1
	}
	return string(sRune)
}

func minusOne(s string, index int) string {
	sRune := []rune(s)
	if sRune[index] == '0' {
		sRune[index] = '9'
	} else {
		sRune[index] -= 1
	}
	return string(sRune)
}
```
