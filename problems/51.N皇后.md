// 最终结果
var result [][]string

// 输⼊棋盘边⻓ n，返回所有合法的放置
func solveNQueens(n int) [][]string {
	result = [][]string{}
	board := make([]string, n)
	for i := 0; i < n; i++ {
		for j := 0; j < n; j++ {
			board[i] += "."
		}
	}
	backtrack(board, 0)
	return result
}

// 路径：board 中⼩于 row 的那些⾏都已经成功放置了皇后
// 选择列表：第 row ⾏的所有列都是放置皇后的选择
// 结束条件：row 超过 board 的最后⼀⾏
func backtrack(board []string, row int) {
	if row == len(board) {
		temp := make([]string, len(board))
		copy(temp, board)
		result = append(result, temp)
		return
	}
	n := len(board[row])
	for col := 0; col < n; col++ {
		// 排除不合法选择
		if !isValid(board, row, col) {
			continue
		}
		// 做选择
		temp := []byte(board[row])
		temp[col] = 'Q'
		board[row] = string(temp)
		// 进⼊下⼀⾏决策
		backtrack(board, row+1)
		// 撤销选择
		temp[col] = '.'
		board[row] = string(temp)
	}
}

func isValid(board []string, row, col int) (b bool) {
	n := len(board)
	// 检查列是否有皇后互相冲突
	for i := 0; i < n; i++ {
		if board[i][col] == 'Q' {
			return false
		}
	}
	// 检查右上⽅是否有皇后互相冲突
	for i, j := row-1, col+1; i >= 0 && j < n; i, j = i-1, j+1 {
		if board[i][j] == 'Q' {
			return false
		}
	}
	// 检查左上⽅是否有皇后互相冲突
	for i, j := row-1, col-1; i >= 0 && j >= 0; i, j = i-1, j-1 {
		if board[i][j] == 'Q' {
			return false
		}
	}
	return true
}
