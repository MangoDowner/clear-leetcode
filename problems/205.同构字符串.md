# 205. 同构字符串
> 原题链接：[205. 同构字符串](https://leetcode-cn.com/problems/isomorphic-strings/)

### 代码
```golang
func isIsomorphic(s string, t string) bool {
	ls := len(s)
	if ls != len(t) {
		return false
	}
	// s中t的映射 / t中s的映射
	m1, m2 := make(map[uint8]uint8), make(map[uint8]uint8)
	for i := 0; i < ls; i++ {
		if m1[s[i]] != 0 {
			if m1[s[i]] != t[i] {
				return false
			}
			continue
		}
		if m2[t[i]] != 0 {
			return false
		}
		m1[s[i]] = t[i]
		m2[t[i]] = s[i]
	}
	return true
}
```
