# 横竖将军里面选矮子
> 原题链接：[807. 保持城市天际线](https://leetcode-cn.com/problems/max-increase-to-keep-city-skyline/)


### 解题思路
思路就是：
```
1、先找出每一行里面最高的建筑
2、再找出每一列里面最高的建筑
3、对于每个单元格来说，他不能超过所在行最高的，也不能超过所在列最高的，
也就是说，两个最高里选个矮的，就OK了
```
### 代码

```golang
func maxIncreaseKeepingSkyline(grid [][]int) int {
	sum := 0
	rowNum := len(grid)
	if rowNum == 0 {
		return sum
	}
	colNum := len(grid[0])
	newGrid := make([][]int, rowNum)
	// 先刷一遍每行的最高建筑
	for row := 0; row < rowNum; row++ {
		max := grid[row][0]
		for col := 1; col < colNum; col++ {
			if grid[row][col] > max {
				max = grid[row][col]
			}
		}
		for col := 0; col < colNum; col++ {
			newGrid[row] = append(newGrid[row], max)
		}
	}
	// 再刷一遍每列的每列的最高建筑
	for col := 0; col < colNum; col++ {
		max := grid[0][col]
		for row := 1; row < rowNum; row++ {
			if grid[row][col] > max {
				max = grid[row][col]
			}
		}
		for row := 0; row < rowNum; row++ {
			// 取值为行列天际线中较矮的
			value := int(math.Min(float64(newGrid[row][col]), float64(max)))
			if value > grid[row][col] {
				sum += value - grid[row][col]
			}
		}
	}
	return sum
}
```
